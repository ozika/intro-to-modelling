rw1_basic <- function(data, params, conf=c()) {
  # rewards
  outcomes = data$outcomes
  trials = data$trials

  vals = matrix(, nrow = data$trial, ncol = data$nCues)
  vals[1,] = rep(0.5, 1, data$nCues)
  

  for (i in seq(data$trial)) {

    if (i != data$nTrials) {
      # pass all 
      vals[i+1,] = vals[i,]
      
      # update cue shown  value
      cue <- data$cue[i]
      vals[i+1,cue] = vals[i,cue] + params$alpha*(alpha[i,cue]  - vals[i,cue])
      
    }
    
  }
  out = list('vals'=vals)
}

show R code for a likelihood function (assuming gaussian likelihood), add documentation to the function below

gaussian_negloglik_fit <- function(x, data, conf = list()) {

  params <- conf$params
  for (i in seq_along(conf$params2estimate)) {
    params[[conf$params2estimate[i]]] <- x[i]
  }

  model_fun <- load_rating_model(
    model_name = conf$model_name,
    model_dir = conf$model_dir %||% file.path("models", "rating_models")
  )

  model_out <- model_fun(data = data, params = params, conf = conf)

  # model prediction given the parameters  
  mu <- clamp01(model_out$vals)

  # participant data 
  y  <- clamp01(data$rating)

  # assumption about likelihood SD
  obs_sd <- conf$obs_sd %||% 0.2


  nll <- -sum(stats::dnorm(y, mean = mu, sd = obs_sd, log = TRUE))
  nll
}
